\begin{algorithm}[ht!]{
  \footnotesize
  \caption{Insertion-Sort$(A)$\label{alg:insertion-sort}}

  \Entrada{Um vetor $A$ de tamanho $n$.}
  \Saida{O vetor $A$ ordenado.}
  
  \Para{$j=2$ \Ate $n$}{
    $key = A[j]$\;
    $i=j-1$\;
    \Enqto{$i>0$ \textnormal{e} $A[i]>key$}{
      $A[i+1] = A[i]$\;
      $i=i-1$\;
    }
    $A[i+1] = key$\;
  }
}
\end{algorithm}
