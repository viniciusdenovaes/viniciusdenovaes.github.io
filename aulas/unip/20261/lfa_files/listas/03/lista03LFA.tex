\item[Operadores de ExpressĂľes Regulares]:
\begin{itemize}
\item A uniĂŁo de duas linguagens $L$ e $M$ denotada por $L\cup M$ 
ĂŠ o conjunto de strings que estĂŁo em $L$ ou em $M$
\item A concatenaĂ§ĂŁo de duas linguagens $L$ e $M$ denotada por $LM$
ĂŠ o conjunto de strings que podem ser formados concatenando-se qualquer string de $L$ 
com qualquer string de $M$, 
nesta ordem
\item A potĂŞncia $L^i$, para $i=0,1,2,3, \dots$; ĂŠ a concatenaĂ§ĂŁo de $L$ com $L^{i-1}$, se $i>0$;
e $L^0 = \{\ve\}$ para $i=0$.
\item O fechamento de uma linguagem $L$, denotado por $L^*$ ĂŠ a uniĂŁo
\[\bigcup_{i=0}^\infty L^i\]
\end{itemize}

\item[DefiniĂ§ĂŁo de uma ExpressĂŁo Regular (ER)]:

\begin{itemize}
\item $\emptyset$ ĂŠ uma ER; $L(\emptyset) = \emptyset$.
\item $\ve$ ĂŠ uma ER; $L(\ve) = \{\ve\}$.
\item \textbf{a} ĂŠ uma ER, $a\in \Sigma$; $L($\textbf{a}$) = \{a\}$.
\item Se $E$ e $F$ sĂŁo ER, entĂŁo $E+F$ ĂŠ ER; $L(E+F) = L(E)\cup L(F)$.
\item Se $E$ e $F$ sĂŁo ER, entĂŁo $EF$ ĂŠ ER; $L(EF) = L(E)L(F)$.
\item Se $E$ ĂŠ ER, entĂŁo $E^*$ ĂŠ ER; $L(E^*) = L(E)^*$.
\item Se $E$ ĂŠ ER, entĂŁo $(E)$ ĂŠ ER; $L((E)) = L(E)$.
\end{itemize}



\end{description}
