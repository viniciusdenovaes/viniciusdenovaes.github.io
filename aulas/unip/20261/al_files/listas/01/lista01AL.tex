    \question Resolva os seguintes sistemas lineares 3D

    \begin{parts}

    \part 

    \[
\left\{
\begin{aligned}
        x + y + z & = 3\\
       -x - y + z & = -1\\
        x - y - z & = -1
\end{aligned}
\right.
\hspace{1in}
\left\{
\begin{aligned}
        x + y + z & = 6\\
       -x - y + z & = -4\\
        x - y - z & = 0
\end{aligned}
\right.
\hspace{1in}
\left\{
\begin{aligned}
        x + y + z & = 2\\
       -x - y + z & = 2\\
        x - y - z & = 0
\end{aligned}
\right.
\]

\vspace{\stretch{1}}


    \part

\[
\left\{
\begin{aligned}
        x + 2y + z & = 4\\
       -x - y  & = 0\\

\end{aligned}
\right.

\left\{
\begin{aligned}
        x + 2y + z & = 8\\
       -x - y  & = -3\\
        -y + 2z & = 4

\right.
\hspace{1in}
\left\{
\begin{aligned}
        x + 2y + z & = 3\\
       -x - y  & = 0\\

\right.
